%% ============================================================
%%  SUBJECT 1 — REAL ANALYSIS
%% ============================================================
\subjectbox{C1}{C1}{1}{Real Analysis}%
{38 Questions across both papers $|$ Part B: 4--5 $|$ Part C: 8--9 $|$ \textbf{Tier 1 — Highest Priority}}

\sectionrule{C1}

\begin{subtopics}

  \item \textbf{Sequences of Real Numbers}
  \begin{subtopics}
    \item \chk Convergence and divergence; Cauchy sequences
    \item \chk $\limsup$ and $\liminf$: definitions, properties, squeeze theorem
    \item \chk Monotone sequences; bounded sequences (Bolzano-Weierstrass)
    \item \chk Subsequences and accumulation points
    \item \chk Decimal expansions of real numbers; digit-by-digit locking property
    \item \chk Sequences defined by recursion: $x_{n+1}=f(x_n)$; convergence to fixed points
  \end{subtopics}

  \item \textbf{Series of Real Numbers}
  \begin{subtopics}
    \item \chk Convergence tests: Ratio, Root (Cauchy-Hadamard), Comparison, Limit Comparison
    \item \chk Alternating series (Leibniz test); absolute vs.\ conditional convergence
    \item \chk Rearrangements of series; Riemann rearrangement theorem
    \item \chk Power series: radius of convergence; term-by-term differentiation/integration
    \item \chk Double series and Cauchy product
    \item \chk Series with log factors: $\sum \frac{1}{n(\log n)^\alpha (\log\log n)^\beta}$ (Bertrand-De Morgan)
  \end{subtopics}

  \item \textbf{Limits and Continuity}
  \begin{subtopics}
    \item \chk $\varepsilon$-$\delta$ definition; one-sided limits; limits at infinity
    \item \chk Uniform continuity: definition, characterisation on compact sets
    \item \chk Composition of uniformly continuous functions (e.g., $f \circ g$ when $g$ is $x^2$ on $(0,1]$)
    \item \chk Intermediate value theorem; fixed-point theorem
    \item \chk Lipschitz functions; relation to uniform continuity
  \end{subtopics}

  \item \textbf{Differentiability}
  \begin{subtopics}
    \item \chk Mean Value Theorem, Cauchy MVT, Taylor's theorem with remainder
    \item \chk L'H\^{o}pital's rule; indeterminate forms
    \item \chk Functions with prescribed derivatives at $0$: $f^{(n)}(0)= n!n^n$ type
    \item \chk Implicit Function Theorem in $\mathbb{R}^n$: existence, uniqueness, derivative formula
    \item \chk Functions of two variables: partial derivatives, differentiability, chain rule
    \item \chk Limits in $\mathbb{R}^2$: path-dependent vs.\ genuine limits; e.g., $(ax+by)^2/(ax^2+by^2)$
  \end{subtopics}

  \item \textbf{Riemann Integration}
  \begin{subtopics}
    \item \chk Riemann integrability: characterisation (Lebesgue criterion — measure of discontinuities is 0)
    \item \chk Fundamental theorem of calculus (both parts)
    \item \chk Improper integrals: convergence on $[0,\infty)$; comparison test
    \item \chk Functions $f \in \mathcal{F}$ (Riemann integrable on $[0,t]$): if $f$ uniformly continuous and $\int_0^\infty f$ converges $\Rightarrow$ $f(x)\to 0$
    \item \chk Existence of $f$ with convergent $\int_0^\infty f$ but $f \not\to 0$; oscillating counterexamples
    \item \chk Monotone functions: one-sided limits always exist; discontinuities are countable
    \item \chk Riemann-Stieltjes integration (basics)
  \end{subtopics}

  \item \textbf{Sequences and Series of Functions}
  \begin{subtopics}
    \item \chk Pointwise vs.\ uniform convergence: definitions, examples, counterexamples
    \item \chk Weierstrass M-test for uniform convergence of series
    \item \chk Continuity under uniform convergence; integration/differentiation term-by-term
    \item \chk Dini's theorem: monotone pointwise $\to$ uniform on compact set
    \item \chk $f_n \to 0$ pointwise but $\int f_n \not\to 0$ examples (tent functions)
    \item \chk Uniform convergence on sub-intervals; behaviour at boundary
  \end{subtopics}

  \item \textbf{Metric Spaces}
  \begin{subtopics}
    \item \chk Metric space axioms; equivalent metrics; discrete metric
    \item \chk Open and closed sets; interior, closure, boundary
    \item \chk Compactness: Heine-Borel, sequential compactness; finite subcover property
    \item \chk Completeness; Cauchy sequences in metric spaces; Baire category theorem
    \item \chk Distance function $d(x,A) = \inf_{a \in A} d(x,a)$: Lipschitz continuity (hence uniform continuity)
    \item \chk Closed set characterisation: $A$ closed $\Leftrightarrow$ $d(x,A)>0$ for all $x \notin A$
    \item \chk Compact metric spaces: continuous image is compact; min/max attained
    \item \chk Contraction mapping principle (Banach Fixed Point Theorem)
  \end{subtopics}

  \item \textbf{Countability and Set Theory}
  \begin{subtopics}
    \item \chk Countable and uncountable sets; Cantor-diagonal argument
    \item \chk Decimal expansions with restricted digits (e.g., digits from $\{0,2,4\}$): uncountability
    \item \chk Continuous injections preserve cardinality; $f(\mathbb{N})$ vs.\ $f(\mathbb{R})$
    \item \chk Finite subsets of $\mathbb{Z}$: countable; power set of $\mathbb{N}$: uncountable
  \end{subtopics}

  \item \textbf{Monotone Functions and Bounded Variation}
  \begin{subtopics}
    \item \chk Monotone increasing on $[0,1]$: one-sided limits exist everywhere; at most countably many discontinuities
    \item \chk Surjective monotone on $[0,1] \Rightarrow$ continuous; Riemann integrable monotone
    \item \chk $U_\alpha=(L_\alpha^-, L_\alpha^+)$ intervals for distinct $\alpha, \beta$ are disjoint
    \item \chk Functions of bounded variation; Jordan decomposition
  \end{subtopics}

\end{subtopics}

%% ============================================================
%%  SUBJECT 2 — LINEAR ALGEBRA
%% ============================================================
\subjectbox{C2}{C2}{2}{Linear Algebra}%
{32 Questions across both papers $|$ Part B: 4 $|$ Part C: 6 $|$ \textbf{Tier 1 — Highest Priority}}

\sectionrule{C2}

\begin{subtopics}

  \item \textbf{Vector Spaces and Subspaces}
  \begin{subtopics}
    \item \chk Subspaces, spanning sets, linear independence
    \item \chk Basis and dimension; Steinitz exchange lemma
    \item \chk Dimension of sum and intersection: $\dim(U+V)=\dim U + \dim V - \dim(U\cap V)$
    \item \chk Quotient space $V/W$; dimension $= \dim V - \dim W$
    \item \chk Spanning set $\{u,v,w,x,y\}$ with $\{u,v,x,y\}$ LI: $\dim V \in \{4,5\}$ argument
  \end{subtopics}

  \item \textbf{Linear Maps and Matrices}
  \begin{subtopics}
    \item \chk Rank-nullity theorem; kernel and image
    \item \chk Matrix representation; change of basis
    \item \chk Invertibility, determinant, adjugate
    \item \chk Trace: $\mathrm{tr}(AB)=\mathrm{tr}(BA)$; trace and eigenvalue sum
    \item \chk Systems of linear equations; row reduction
  \end{subtopics}

  \item \textbf{Eigenvalues and Eigenvectors}
  \begin{subtopics}
    \item \chk Characteristic polynomial; Cayley-Hamilton theorem
    \item \chk Newton's power-sum identities: $\mathrm{tr}(A)=\sum\lambda_i$, $\mathrm{tr}(A^2)=\sum\lambda_i^2$; char poly from trace data
    \item \chk Algebraic and geometric multiplicity; defective matrices
    \item \chk Eigenvalues of functions: $f(A)$ has eigenvalues $f(\lambda_i)$; Cayley-Hamilton application
    \item \chk Largest real eigenvalue of a specific $4\times 4$ matrix (e.g., sum of rows trick)
    \item \chk Matrices over $\mathbb{C}$: trace 1, det 1, one eigenvalue = 1 $\Rightarrow$ structure of other two
  \end{subtopics}

  \item \textbf{Diagonalisation and Jordan Normal Form}
  \begin{subtopics}
    \item \chk Diagonalisability criteria: $n$ distinct eigenvalues; geom.\ mult.\ = alg.\ mult.
    \item \chk If $T^2$ is diagonalisable, when is $T$ diagonalisable? (need non-negative eigenvalues of $T^2$)
    \item \chk If $T^2$ has only negative eigenvalues $\Rightarrow$ $\dim V$ is even
    \item \chk Jordan blocks; minimal polynomial computation
    \item \chk Simultaneous diagonalisation of commuting matrices
  \end{subtopics}

  \item \textbf{Inner Product Spaces}
  \begin{subtopics}
    \item \chk Inner product axioms; Cauchy-Schwarz; triangle inequality
    \item \chk Gram-Schmidt orthogonalisation; orthonormal bases
    \item \chk Orthogonal complement; projection theorem; best approximation
    \item \chk Adjoint of a linear map; self-adjoint, unitary, normal operators
    \item \chk $M_n(\mathbb{C})$ with $\langle A,B\rangle = \mathrm{tr}(A^*B)$; rank-$k$ conditions; $W_k=\{0\}$ for $k\ge 1$
    \item \chk Subspace of $C[0,\pi]$ spanned by $\{f_1,f_2,f_3\}$ with inner product $\frac{2}{\pi}\int f g \sin^2 x\,dx$
    \item \chk FALSE: ``eigenvectors of $B$ for distinct eigenvalues are $B$-orthogonal'' (not true with non-standard inner product $\langle v,w\rangle=v^\top Bw$)
  \end{subtopics}

  \item \textbf{Quadratic Forms}
  \begin{subtopics}
    \item \chk Positive/negative definite, semi-definite, indefinite; Sylvester's criterion
    \item \chk Sylvester's law of inertia; signature; congruence
    \item \chk Quadratic form $Q(f)=\int_{-1}^1(f')^2dt$ on $V=\{ax^3+bx^2+cx\}$: FALSE statements
    \item \chk Complex quadratic forms over $\mathbb{Q}$; zeros over $\mathbb{C}$ vs $\mathbb{R}$ vs $\mathbb{Q}$
  \end{subtopics}

  \item \textbf{Projections and Special Operators}
  \begin{subtopics}
    \item \chk Idempotent operators: $T^2=T$; direct sum decomposition $V=\ker T \oplus \mathrm{im}\, T$
    \item \chk $T^2=T$: $W=\mathrm{span}\{I,T^n:n\ge 1\}$ has $\dim W=2$; idempotents in $W$
    \item \chk If $U^2=U$ and $(T-U)^2=T-U$ $\Rightarrow$ $(TU)^2=TU$
    \item \chk $A^2=I$, $\mathrm{Tr}(A)=0$: anticommutant $\{B:AB+BA=0\}$ is 8-dimensional $\mathbb{C}$-vector space
    \item \chk Commutant $\{B:AB=BA\}$ also 8-dimensional; existence of $B$ with $B^2=I$, $\mathrm{Tr}(B)=0$, $AB+BA=0$
    \item \chk Existence of 3-dimensional invariant subspace for any operator on 4-dim $\mathbb{C}$ space
  \end{subtopics}

  \item \textbf{Matrices over Finite Fields}
  \begin{subtopics}
    \item \chk $GL_n(\mathbb{F}_q)$: number of invertible matrices = $(q^n-1)(q^n-q)\cdots(q^n-q^{n-1})$
    \item \chk Number of $2\times 2$ matrices over $\mathbb{F}_5$ of rank 1: subtract invertible + zero from total $5^4$
    \item \chk Subspaces of $\mathbb{F}_q^{2n}$ containing a given $n$-dim subspace (Grassmannian counting)
  \end{subtopics}

  \item \textbf{Integer Matrices and Minimal Polynomials}
  \begin{subtopics}
    \item \chk Non-zero $3\times 3$ integer matrix: char poly over $\mathbb{Q}$; if $f(\lambda_i)=0$ for all $i$ and $f$ cubic $\Rightarrow f(A)=0$ (Cayley-Hamilton suffices)
    \item \chk Does $f \in \mathbb{Q}[X]$ quadratic exist with all eigenvalues as roots? Only if poly factors
    \item \chk Minimal polynomial divides any annihilating polynomial
  \end{subtopics}

\end{subtopics}

%% ============================================================
%%  SUBJECT 3 — STATISTICS & PROBABILITY
%% ============================================================
\subjectbox{C3}{C3}{3}{Statistics \& Probability}%
{30 Questions across both papers $|$ Part B: 5 $|$ Part C: 8 $|$ \textbf{Tier 1 — Highest Priority}}

\sectionrule{C3}

\begin{subtopics}

  \item \textbf{Point Estimation}
  \begin{subtopics}
    \item \chk Maximum likelihood estimation (MLE): discrete and continuous families
    \item \chk Method of moments (MOM): equate population moments to sample moments
    \item \chk Sufficient statistics: Factorisation theorem (Neyman-Fisher)
    \item \chk Minimal sufficient statistics; completeness
    \item \chk Ancillary statistics: definition; examples (ratios, differences of normals)
    \item \chk Unbiased estimators; UMVUE via Lehmann-Scheff\'{e}; Rao-Blackwellisation
    \item \chk Cram\'{e}r-Rao lower bound; Fisher information
    \item \chk Consistency: $\hat\theta_n \xrightarrow{p} \theta$; MLE under regularity is consistent
  \end{subtopics}

  \item \textbf{Hypothesis Testing}
  \begin{subtopics}
    \item \chk Neyman-Pearson lemma; most powerful tests; UMP tests
    \item \chk Likelihood ratio test (LRT); Wilks' theorem ($-2\log\Lambda \xrightarrow{d}\chi^2$)
    \item \chk Type I / Type II errors; significance level; $p$-value
    \item \chk One-sided and two-sided tests; uniformly most powerful (UMP) tests
    \item \chk Sign test: $K=$ number of obs $>$ $\theta_0$; $K\sim\text{Bin}(n,1/2)$ under $H_0$
    \item \chk Non-parametric tests: Wilcoxon, Mann-Whitney (basics)
    \item \chk Test statistic $F$ for regression significance; relation to $R^2$
    \item \chk Confidence intervals from pivot quantities; relation to hypothesis tests
  \end{subtopics}

  \item \textbf{Bayesian Estimation}
  \begin{subtopics}
    \item \chk Prior, likelihood, posterior: Bayes' theorem in density form
    \item \chk Beta-Binomial conjugate: prior Beta$(\alpha,\beta)$ + Binomial$(n,\theta)$ $\Rightarrow$ posterior Beta$(\alpha+x, \beta+n-x)$
    \item \chk Posterior mean $= \frac{x+\alpha}{n+\alpha+\beta}$; posterior mean $\gtrless$ prior mean $\Leftrightarrow$ $x/n \gtrless \alpha/(\alpha+\beta)$
    \item \chk Improper prior $\pi(\theta)\propto 1$: posterior is proportional to likelihood
    \item \chk Normal-Normal conjugate: $X|\theta\sim N(\theta,\sigma^2)$, prior $\theta\sim N(\mu_0,\tau^2)$; posterior mean is weighted average
    \item \chk Posterior mode vs.\ MLE; posterior variance vs.\ $\mathrm{Var}(X|\theta)$
  \end{subtopics}

  \item \textbf{Probability Distributions}
  \begin{subtopics}
    \item \chk Exponential, Gamma, Beta, Normal, Binomial, Poisson, Uniform, Cauchy
    \item \chk MGF identification: $M_X(t)=e^{3(e^t-1)} \Rightarrow X\sim\text{Poisson}(3)$; $M_Y(t)=(\tfrac{1}{2}e^{-3t}+\tfrac{1}{2}e^{3t})^2$ etc.
    \item \chk CDF, survival function $S(t)$, hazard rate $\lambda(t)=f(t)/S(t)$, cumulative hazard $H(t)$
    \item \chk Series system lifetime: $Y=\min\{T_1,\ldots,T_n\}$; $P(Y>t)=(S(t))^n$
    \item \chk Reliability / survival analysis: $\lambda(t)=1/(1+t) \Rightarrow H(t)=\ln(1+t)$, $S(t)=1/(1+t)$
    \item \chk Mixture distributions (brand I exponential + brand II gamma)
    \item \chk Discrete distribution with PMF $P(X=k)=\theta^k/k!$ type; MLE from observed counts
  \end{subtopics}

  \item \textbf{Convergence and Limit Theorems}
  \begin{subtopics}
    \item \chk Convergence in distribution, probability, almost sure, $L^p$; hierarchy
    \item \chk Weak Law of Large Numbers; Strong LLN (Kolmogorov)
    \item \chk Central Limit Theorem; CLT for i.i.d.\ with finite variance
    \item \chk Extreme value theory: $T_n=\max\{X_1,\ldots,X_n\}-\ln n \xrightarrow{d}$ Gumbel distribution
    \item \chk Gumbel CDF: $F(x)=e^{-e^{-x}}$; $P(Y\le 0)=e^{-1}$; median $=\ln(\ln 2)$
    \item \chk Remainder $X_n = Y_n \bmod m$: as $n\to\infty$, converges to Uniform on $\{0,1,\ldots,m-1\}$
  \end{subtopics}

  \item \textbf{Multivariate Distributions}
  \begin{subtopics}
    \item \chk Multivariate normal $N_p(\mu,\Sigma)$: marginals, conditionals, affine transformations
